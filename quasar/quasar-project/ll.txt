<script setup lang="ts">
// import type { Todo, Meta } from 'components/models';

import { ref, onMounted } from 'vue';
import { api } from 'src/boot/axios';
import { useQuasar } from 'quasar';
const image = ref('');
const birth = ref([]); // ✅ สร้าง state

const $q = useQuasar();

if ($q.screen.lt.md) {
  console.log('มือถือ');
}
const getImageUrl = (image: string) => {
  return api(`/file/${image}`, {
    responseType: 'blob', // รับข้อมูลเป็น Blob
  })
    .then((response) => {
      return URL.createObjectURL(response.data); // สร้าง URL จาก Blob
    })
    .catch((error) => {
      console.error('Error fetching image:', error);
      return ''; // คืนค่าเป็น string ว่างในกรณีเกิดข้อผิดพลาด
    });
};
const fetchBirthCard = async () => {
  try {
    const response = await api.get('/festival/all');
    console.log('Birth Card Data:', response.data.festival[0]);
    birth.value = response.data.festival[0]; // ✅ ใช้ .value
    image.value = response.data.festival[0]?.image;
    if (image.value) {
      image.value = await getImageUrl(image.value); // ✅ อัปเดต image ด้วย URL ที่ได้จาก getImageUrl
      console.log(image.value);
    }
  } catch (error) {
    console.error('Error fetching birth card data:', error);
  }
};
onMounted(() => {
  void fetchBirthCard();
});
</script>
