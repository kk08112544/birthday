<script setup lang="ts">
import { ref } from 'vue';
import { useQuasar } from 'quasar';
import { api } from 'src/boot/axios';
import { useRouter } from 'vue-router';

const $q = useQuasar();
const router = useRouter();

// สร้างตัวแปรมารับค่าจากฟอร์ม
const festivalName = ref('');
// const wishWord = ref('');
const imageFile = ref<File | null>(null);
// const imageCardFile = ref<File | null>(null);
const loading = ref(false);

// const showWishDialog = ref(false);
const showCardDialog = ref(false);

const AddWishDialog = ref(false);

// เพิ่มตัวแปรเหล่านี้ใน <script setup>
const tempCardFile = ref<File | null>(null);
const cardFileList = ref<File[]>([]);

// ฟังก์ชันเพิ่มการ์ดเข้า List
const addCardToList = () => {
  if (tempCardFile.value) {
    cardFileList.value.push(tempCardFile.value);
    tempCardFile.value = null; // ล้างค่า
    showCardDialog.value = false; // ปิด Dialog
  }
};

// ฟังก์ชันลบการ์ด
const removeCard = (index: number) => {
  cardFileList.value.splice(index, 1);
};

// ฟังก์ชันสำหรับ Preview รูป (ถ้าต้องการโชว์รูปเล็กๆ)
const getFilePreview = (file: File) => {
  return URL.createObjectURL(file);
};
// เก็บค่าที่กำลังกรอกใน Dialog
const tempWish = ref('');
// const tempCardImage = ref<File | null>(null);
// เก็บรายการคำอวยพรทั้งหมด (เพื่อส่งไป Backend)
const wishWordList = ref<string[]>([]);

// ฟังก์ชันเมื่อกด "เพิ่มคำอวยพร" ใน Dialog
// const addWishToList = () => {
//   if (tempWish.value.trim() !== '') {
//     wishWordList.value.push(tempWish.value);
//     tempWish.value = ''; // ล้างค่าที่กรอก
//     showWishDialog.value = false; // ปิด Dialog
//   }
// };
const addWishToList = () => {
  if (tempWish.value.trim() !== '') {
    wishWordList.value.push(tempWish.value);
    tempWish.value = ''; // ล้างค่าที่กรอก
    AddWishDialog.value = false; // ปิด Dialog
  }
};

// ฟังก์ชันลบคำอวยพร (ถ้ากดผิด)
const removeWish = (index: number) => {
  wishWordList.value.splice(index, 1);
};

const addFestival = async () => {
  loading.value = true;
  const accessToken = localStorage.getItem('accessToken');

  try {
    // 1. อัปโหลดรูปภาพหน้าปก (Main Image)
    let festivalImageName = '';
    if (imageFile.value) {
      const formData = new FormData();
      formData.append('singleFile', imageFile.value);
      const res = await api.post('/upload', formData);
      festivalImageName = res.data.image;
    }

    // 2. อัปโหลดรูปการ์ด (Multiple Cards)
    // ใช้ Promise.all เพื่อให้อัปโหลดพร้อมกัน (Parallel) จะเร็วกว่า loop await ทีละตัว
    let cardImageNames = [];
    if (cardFileList.value && cardFileList.value.length > 0) {
      const uploadPromises = cardFileList.value.map(async (file) => {
        const formData = new FormData();
        formData.append('singleFile', file);
        const res = await api.post('/upload', formData);
        return res.data.image; // คืนค่าชื่อไฟล์กลับไปใน Array
      });

      // รอให้อัปโหลดเสร็จทุกไฟล์
      cardImageNames = await Promise.all(uploadPromises);
    }

    // 3. เตรียม Payload
    const payload = {
      festivalName: festivalName.value,
      image: festivalImageName,
      // แปลงเป็น Array of Objects ตามที่คุณต้องการ
      wisher: wishWordList.value.map((word) => ({ wishWord: word })),
      card: cardImageNames.map((imgName) => ({ imageCard: imgName })),
    };

    // 4. บันทึกข้อมูลลงฐานข้อมูล
    const response = await api.post('/admin/festival', payload, {
      headers: { Authorization: `Bearer ${accessToken}` },
    });

    $q.notify({ color: 'positive', message: response.data.message, icon: 'check' });
    await router.push('/admin/festival');
  } catch (error) {
    console.error('เกิดข้อผิดพลาด:', error);
    $q.notify({
      color: 'negative',
      message: 'บันทึกไม่สำเร็จ: ',
      icon: 'report_problem',
    });
  } finally {
    loading.value = false;
  }
};
</script>
