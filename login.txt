<script setup lang="ts">
import { ref } from 'vue';
import { useQuasar } from 'quasar';
import { useRouter } from 'vue-router';
import { api } from 'src/boot/axios';
import type { AxiosError } from 'axios';

const $q = useQuasar();
const router = useRouter();

const userName = ref('');
const password = ref('');
const isPwd = ref(true);
const loading = ref(false);

const onSubmit = async () => {
  loading.value = true;
  try {
    const response = await api.post('/auth/login', {
      userName: userName.value,
      password: password.value,
    });

    const { accessToken, refreshToken, uId, firstName, userName: uName } = response.data.user.user;

    localStorage.setItem('accessToken', accessToken);
    localStorage.setItem('refreshToken', refreshToken);
    localStorage.setItem('userId', uId);
    localStorage.setItem('firstName', firstName);
    localStorage.setItem('username', uName);

    if (response.status === 201) {
      $q.notify({
        color: 'positive',
        textColor: 'white',
        icon: 'check_circle',
        message: response.data.message || 'เข้าสู่ระบบสำเร็จ',
        position: 'top',
        timeout: 2000,
      });
      router.push('/admin/festival').catch(console.error);
    }
  } catch (err: unknown) {
    const error = err as AxiosError<{ message: string }>;
    const errorResponse = error.response;

    $q.notify({
      color: 'negative',
      message:
        errorResponse?.status === 404
          ? (errorResponse.data?.message ?? 'ไม่พบผู้ใช้งาน')
          : 'เกิดข้อผิดพลาดในการเข้าสู่ระบบ',
      icon: 'error',
      position: 'top',
    });
  } finally {
    loading.value = false;
  }
};
</script>
